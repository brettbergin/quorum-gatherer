1"""Providers settings page — progressive staged flow with per-role model selection. 

2 

3Select a provider, paste its key, then **Test key & load models**: that one call authenticates the 

4key *and* fetches the live model catalog the account actually has access to. Models are then chosen 

5*per council role* — the deliberating council members and the synthesizing Chairman get their own 

6model (and, where the model supports it, a provider-specific reasoning knob). This keeps the 

7Chairman on a capable synthesis model instead of being downgraded with the members. **Save** 

8persists. 

9"""

86 def configure(self, spec: dict | None) -> None: 

87 """Set the provider spec and lay out the matching reasoning variant.""" 

88 self._spec = spec 

89 kind = self._kind() 

90 has_reasoning = kind != "none" 

91 self._reasoning_row.setVisible(has_reasoning) 

92 is_effort = kind == "effort" 

93 is_budget = kind == "thinking_budget" 

94 self._effort.setVisible(is_effort) 

95 self._budget_enable.setVisible(is_budget) 

96 self._budget.setVisible(is_budget) 

97 

98 if is_effort and spec is not None: 

99 self._effort.blockSignals(True) 

100 self._effort.clear() 

101 self._effort.addItem("Off", "") 

102 for e in spec["reasoning"]["efforts"]: 

103 self._effort.addItem(e, e) 

104 self._effort.blockSignals(False) 

105 elif is_budget and spec is not None: 

106 r = spec["reasoning"] 

107 self._budget.setRange(max(1, r["budget_min"]), r["budget_max"]) 

108 self._budget.setValue(r["budget_default"] or max(1, r["budget_min"]))

110 def preview_stored(self, stored_model: str | None, stored_reasoning: str | None) -> None: 

111 """Before a catalog is loaded, show the currently-configured model (locked).""" 

112 self._models = [] 

113 self._locked = True

116 if stored_model: 

117 self.model.addItem(stored_model, stored_model) 

118 self.model.setCurrentIndex(0)

120 self.model.blockSignals(False) 

121 self._apply_stored_reasoning(stored_reasoning)

124 def load_catalog( 

125 self, models: list[dict], stored_model: str | None, stored_reasoning: str | None 

126 ) -> None: 

127 """After Test, populate from the live catalog and pre-select the stored model.""" 

128 self._models = models 

129 self._locked = False 

130 self.model.blockSignals(True) 

131 self.model.clear() 

132 for m in models: 

133 label = m["label"] if m["label"] == m["id"] else f"{m['label']} ({m['id']})" 

134 self.model.addItem(label, m["id"]) 

135 self.model.setEnabled(bool(models)) 

136 if stored_model: 

137 idx = self.model.findData(stored_model) 

138 if idx >= 0: 

139 self.model.setCurrentIndex(idx) 

140 self.model.blockSignals(False) 

141 self._apply_stored_reasoning(stored_reasoning) 

142 self._refresh_reasoning()

148 def reasoning_raw(self) -> str: 

149 """Raw reasoning value to persist — "" clears it.""" 

150 kind = self._kind() 

151 model = self._selected_model() 

152 if kind == "none" or not (model and model.get("supports_reasoning")): 

153 return "" 

154 if kind == "effort": 

155 return self._effort.currentData() or "" 

156 if kind == "thinking_budget": 

157 return str(self._budget.value()) if self._budget_enable.isChecked() else "" 

158 return ""

160 # --- internals --------------------------------------------------------- 

161 def _kind(self) -> str: 

162 return (self._spec or {}).get("reasoning", {}).get("kind", "none") 

163 

164 def _selected_model(self) -> dict | None: 

165 mid = self.model.currentData() 

166 if not mid: 

167 return None 

168 found = next((m for m in self._models if m["id"] == mid), None) 

169 if found is not None: 

170 return found 

171 # Catalog not loaded yet: synthesize so reasoning reflects the stored model's capability. 

172 pattern = (self._spec or {}).get("reasoning", {}).get("model_pattern", "") 

173 supports = bool(pattern) and re.search(pattern, mid, re.IGNORECASE) is not None 

174 return {"id": mid, "label": mid, "supports_reasoning": supports} 

175 

176 def _apply_stored_reasoning(self, stored: str | None) -> None: 

177 kind = self._kind() 

178 if kind == "effort": 

179 idx = self._effort.findData(stored) if stored else 0 

180 self._effort.setCurrentIndex(idx if idx >= 0 else 0) 

181 elif kind == "thinking_budget": 

182 if stored and stored.isdigit(): 

183 self._budget_enable.setChecked(True) 

184 self._budget.setValue(int(stored)) 

185 else: 

186 self._budget_enable.setChecked(False) 

187 

188 def _refresh_reasoning(self) -> None: 

189 """Enable the reasoning control only when the model supports it and we're unlocked.""" 

190 kind = self._kind() 

191 if kind == "none": 

192 return 

193 model = self._selected_model() 

194 supported = bool(model and model.get("supports_reasoning")) 

195 active = supported and not self._locked 

196 if kind == "effort": 

197 self._effort.setEnabled(active) 

198 if not supported: 

199 self._effort.setCurrentIndex(0) 

200 elif kind == "thinking_budget": 

201 self._budget_enable.setEnabled(active) 

202 self._budget.setEnabled(active and self._budget_enable.isChecked()) 

203 if not supported: 

204 self._budget_enable.setChecked(False) 

205 self._hint.setText("" if supported else "No reasoning for this model.")

304 async def _load(self) -> None: 

305 self._state = await engine.provider_state() 

306 self._provider.blockSignals(True) 

307 self._provider.clear() 

308 for spec in self._state["specs"]: 

309 s = self._state["settings"].get(spec["key"]) 

310 tag = ( 

311 " ✓ enabled" 

312 if s and s["is_enabled"] 

313 else " (disabled)" 

314 if s and s["has_key"] 

315 else "" 

316 ) 

317 self._provider.addItem(spec["label"] + tag, spec["key"]) 

318 self._provider.blockSignals(False) 

319 self._on_provider_changed()

321 def _on_provider_changed(self) -> None: 

322 spec = self._current_spec() 

323 setting = self._current_setting() or {} 

324 self._models = [] 

325 self._key.clear() 

326 self._key.setPlaceholderText( 

327 "•••••••• stored — leave blank to reuse" if setting.get("has_key") else "Paste API key" 

328 ) 

329 # Models + reasoning stay locked until the key is tested; show the configured models. 

330 self._members.configure(spec) 

331 self._members.preview_stored(setting.get("default_model"), setting.get("reasoning")) 

332 self._chairman.configure(spec) 

333 self._chairman.preview_stored( 

334 setting.get("chairman_model"), setting.get("chairman_reasoning") 

335 ) 

336 self._save.setEnabled(False) 

337 

338 if setting.get("is_enabled"): 

339 self._set_badge("enabled", "enabled") 

340 elif setting.get("has_key"): 

341 self._set_badge("disabled", "disabled") 

342 else: 

343 self._set_badge("not configured", "") 

344 self._disable.setEnabled(bool(setting.get("has_key"))) 

345 self._set_status("Test the key to load this provider's models." if spec else "", "")

358 async def _do_test(self) -> None: 

359 self._test.setEnabled(False) 

360 self._set_status("Validating key and loading models…", "applying") 

361 provider = self._provider.currentData() 

362 try: 

363 self._models = await engine.fetch_models(provider, self._key.text() or None) 

364 except Exception as exc: # noqa: BLE001 - surface provider/auth errors to the user 

365 self._test.setEnabled(True) 

366 self._models = [] 

367 self._save.setEnabled(False) 

368 self._set_status(f"⚠ {self._short(exc)}", "error") 

369 return 

370 self._test.setEnabled(True) 

371 

372 setting = self._current_setting() or {} 

373 self._members.load_catalog( 

374 self._models, setting.get("default_model"), setting.get("reasoning") 

375 ) 

376 self._chairman.load_catalog( 

377 self._models, setting.get("chairman_model"), setting.get("chairman_reasoning") 

378 ) 

379 self._save.setEnabled(bool(self._models)) 

380 

381 if self._models: 

382 self._set_badge("key valid", "valid") 

383 self._set_status(f"✓ Key valid — {len(self._models)} models loaded.", "ok") 

384 else: 

385 self._set_status("Key valid, but no models were returned.", "error")

387 async def _do_save(self) -> None: 

388 self._save.setEnabled(False) 

389 self._set_status("Validating and saving…", "applying") 

390 provider = self._provider.currentData() 

391 ok, error = await engine.apply_provider( 

392 provider, 

393 self._key.text() or None, 

394 self._members.model_id(), 

395 self._members.reasoning_raw(), 

396 self._chairman.model_id(), 

397 self._chairman.reasoning_raw(), 

398 ) 

399 self._save.setEnabled(True) 

400 if ok: 

401 await self._load() 

402 self._set_status("✓ Saved — provider enabled.", "ok") 

403 else: 

404 self._set_status(f"⚠ {error}", "error")

406 async def _do_disable(self) -> None: 

407 provider = self._provider.currentData() 

408 await engine.disable_provider(provider) 

409 await self._load() 

410 self._set_status("Provider disabled.", "")
