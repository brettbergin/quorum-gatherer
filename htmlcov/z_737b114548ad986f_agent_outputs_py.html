1"""Structured outputs the agents are forced to produce (pydantic-ai output types). 

2 

3- Council members (Phase A) return `CouncilContribution`. 

4- The Chairman (Phase B) returns `CouncilReportContent` — the full Output Format. 

5""" 

6 

7from __future__ import annotations 

8 

9from typing import Literal 

10 

11from pydantic import BaseModel, Field 

12 

13# --------------------------------------------------------------------------- members 

14 

15 

16class CouncilContribution(BaseModel): 

17 """One council member's independent contribution during deliberation.""" 

18 

19 headline: str = Field(description="One-line stance from this member's point of view.") 

20 perspective: str = Field(description="Concise, non-redundant analysis from the role.") 

21 key_points: list[str] = Field(default_factory=list) 

22 concerns: list[str] = Field(default_factory=list) 

23 # Owned Output-Format sections this member drives, keyed by section id 

24 # (e.g. {"investment_review": "...", "stress_feasibility": "..."}). 

25 section_contributions: dict[str, str] = Field(default_factory=dict) 

26 

27 

28# --------------------------------------------------------------------------- chairman 

29 

30 

31class ExecutiveSummary(BaseModel): 

32 recommendation: str 

33 why_now: str 

34 biggest_risk: str 

35 success_metric: str 

36 next_action: str 

37 

38 

39class KeyAssumptions(BaseModel): 

40 facts: list[str] = Field(default_factory=list) 

41 assumptions: list[str] = Field(default_factory=list) 

42 speculation: list[str] = Field(default_factory=list) 

43 

44 

45class Alternative(BaseModel): 

46 name: str 

47 strengths: str 

48 weaknesses: str 

49 comparison: str = Field(description="How it compares to the original idea.") 

50 

51 

52class StressTest(BaseModel): 

53 customer_value: str 

54 business_value: str 

55 strategic_differentiation: str 

56 feasibility: str 

57 distribution: str 

58 timing: str 

59 defensibility: str 

60 

61 

62class InvestmentReview(BaseModel): 

63 recommendation: Literal["Invest", "Investigate", "Delay", "Reject"] 

64 opportunity_cost: str 

65 why_this_wins: str 

66 

67 

68class DecisionClassification(BaseModel): 

69 reversible: list[str] = Field(default_factory=list) 

70 difficult_to_reverse: list[str] = Field(default_factory=list) 

71 

72 

73class SuccessMetrics(BaseModel): 

74 leading: list[str] = Field(default_factory=list) 

75 lagging: list[str] = Field(default_factory=list) 

76 

77 

78class FinalRecommendation(BaseModel): 

79 recommendation: str 

80 confidence: Literal["High", "Medium", "Low"] 

81 evidence_quality: Literal["Strong", "Moderate", "Weak"] 

82 rationale: str 

83 next_actions: list[str] = Field(default_factory=list) 

84 

85 

86class CouncilReportContent(BaseModel): 

87 """The Chairman's full synthesis — mirrors the source prompt's Output Format.""" 

88 

89 executive_summary: ExecutiveSummary 

90 clarified_idea: str 

91 strategic_question: str 

92 key_assumptions: KeyAssumptions 

93 alternative_approaches: list[Alternative] = Field(default_factory=list) 

94 stress_test: StressTest 

95 investment_review: InvestmentReview 

96 decision_classification: DecisionClassification 

97 mvp: str 

98 validation_experiments: list[str] = Field(default_factory=list) 

99 success_metrics: SuccessMetrics 

100 kill_criteria: list[str] = Field(default_factory=list) 

101 tensions: list[str] = Field( 

102 default_factory=list, description="Decision-relevant disagreements / trade-offs." 

103 ) 

104 final_recommendation: FinalRecommendation 

105 

106 

107# Registry so agent definition files can reference an output schema by name. 

108OUTPUT_SCHEMAS: dict[str, type[BaseModel]] = { 

109 "CouncilContribution": CouncilContribution, 

110 "CouncilReportContent": CouncilReportContent, 

111} 

112 

113 

114def resolve_output_schema(name: str) -> type[BaseModel]: 

115 try: 

116 return OUTPUT_SCHEMAS[name] 

117 except KeyError as exc: 

118 raise KeyError(f"unknown output_schema '{name}'; known: {sorted(OUTPUT_SCHEMAS)}") from exc
