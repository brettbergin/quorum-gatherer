20def _row_from_file(path: Path) -> AgentConfig: 

21 data, body = parse_frontmatter(path.read_text(encoding="utf-8"), source=path.name) 

22 # Validate/normalize through AgentDefinition so defaults match the file-based loader exactly. 

23 defn = AgentDefinition(**data, system_prompt=body, charter="") 

24 return AgentConfig( 

25 key=defn.key, 

26 name=defn.name, 

27 role=str(defn.role), 

28 phase=str(defn.phase), 

29 default_provider=defn.default_provider, 

30 default_model=defn.default_model, 

31 temperature=defn.temperature, 

32 display_order=defn.order, 

33 owned_sections=list(defn.owned_sections), 

34 output_schema=defn.output_schema, 

35 system_prompt=defn.system_prompt, 

36 baseline_system_prompt=defn.system_prompt, 

37 baseline_meta=data, 

38 is_builtin=True, 

39 )

42async def seed_agents(session: AsyncSession, directory: str | Path | None = None) -> int: 

43 """Insert shipped agents into `agent_configs` if the table is empty. Returns rows inserted. 

44 

45 Idempotent: does nothing once any agent rows exist. Caller commits. 

46 """ 

47 existing = await session.scalar(select(func.count()).select_from(AgentConfig)) 

48 if existing: 

49 return 0 

50 

51 settings = get_settings() 

52 directory = Path(directory) if directory else settings.agent_prompts_dir 

53 if not directory.is_dir(): 

54 return 0 

55 

56 rows = [_row_from_file(p) for p in sorted(directory.glob(AGENT_GLOB))] 

57 for row in rows: 

58 session.add(row) 

59 await session.flush() 

60 return len(rows)
