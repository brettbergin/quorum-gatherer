37def _spawn(coro) -> None: 

38 task = asyncio.ensure_future(coro) 

39 task.add_done_callback(lambda t: t.exception() if not t.cancelled() else None)

135 def _rebuild_grid(self) -> None: 

136 """(Re)build the agent panel grid from the current roster. Called after the agents are 

137 loaded/seeded and whenever the roster changes, so new agents appear without a restart.""" 

138 while self._grid.count(): 

139 item = self._grid.takeAt(0) 

140 if item is None: 

141 continue 

142 w = item.widget() 

143 if w is not None: 

144 w.deleteLater() 

145 self._panels = {} 

146 for i, member in enumerate(engine.registry().members): 

147 panel = AgentPanel(member.name, member.key) 

148 panel.details_requested.connect(lambda key: _spawn(self._open_agent_chat(key))) 

149 panel.active_toggled.connect(self._on_agent_toggled) 

150 self._panels[member.key] = panel 

151 self._grid.addWidget(panel, i // 3, i % 3)

154 async def _startup(self) -> None: 

155 await engine.ensure_agents() # first-boot ETL of prompts into the DB; build registry 

156 self._rebuild_grid() 

157 chats = await engine.list_chats() 

158 if not chats: 

159 cid = await engine.create_chat("New council session", None) 

160 chats = await engine.list_chats() 

161 else: 

162 cid = chats[0]["id"] 

163 self._refresh_chat_list(chats) 

164 await self._select_chat(cid)

166 def _refresh_chat_list(self, chats: list[dict]) -> None: 

167 self._chat_titles = {c["id"]: (c["title"] or "") for c in chats} 

168 self._chat_list.clear() 

169 for c in chats: 

170 item = QListWidgetItem(f"{c['title'] or 'Untitled'}\n{c['status']}") 

171 item.setData(Qt.ItemDataRole.UserRole, c["id"]) 

172 self._chat_list.addItem(item)

174 self._chat_list.setCurrentItem(item)

181 # ----------------------------------------------------------------- rename / delete 

182 # Modal dialogs (menu.exec / QInputDialog / QMessageBox) run a NESTED Qt event loop, during 

183 # which qasync does not drive asyncio. So all prompting happens in these synchronous slots; only 

184 # the DB work is spawned afterwards, as a clean task with no nested loop inside it. 

185 def _on_chat_context_menu(self, pos) -> None: 

186 item = self._chat_list.itemAt(pos) 

187 if item is None: 

188 return 

189 chat_id = item.data(Qt.ItemDataRole.UserRole) 

190 menu = QMenu(self._chat_list) 

191 rename_action = menu.addAction("Rename…") 

192 delete_action = menu.addAction("Delete") 

193 chosen = menu.exec(self._chat_list.mapToGlobal(pos)) 

194 if chosen == rename_action: 

195 self._prompt_rename(chat_id) 

196 elif chosen == delete_action: 

197 self._confirm_delete(chat_id)

199 def _prompt_rename(self, chat_id: str) -> None: 

200 # Non-blocking dialog (show, not exec): a task spawned right after a modal's nested event 

201 # loop collapses is never driven by qasync, so we trigger the async work from a signal 

202 # that fires on a clean outer-loop tick instead. 

203 dlg = QInputDialog(self) 

204 dlg.setWindowTitle("Rename session") 

205 dlg.setLabelText("Session name:") 

206 dlg.setTextValue(self._chat_titles.get(chat_id, "")) 

207 dlg.setModal(True) 

208 dlg.textValueSelected.connect(lambda text: _spawn(self._do_rename(chat_id, text))) 

209 dlg.finished.connect(lambda _r: dlg.deleteLater()) 

210 self._session_dialog = dlg 

211 dlg.show()

231 async def _do_rename(self, chat_id: str, title: str) -> None: 

232 await engine.rename_chat(chat_id, title) 

233 self._refresh_chat_list(await engine.list_chats()) 

234 if chat_id == self._chat_id: 

235 self._title.setText(title.strip() or "Council session")

237 async def _do_delete(self, chat_id: str) -> None: 

238 await engine.delete_chat(chat_id) 

239 chats = await engine.list_chats() 

240 if not chats: 

241 await self._new_session() 

242 return 

243 # If the active session was deleted, switch to another one first. 

244 if chat_id == self._chat_id: 

245 await self._select_chat(chats[0]["id"]) 

246 # Always rebuild the sidebar so the deleted row disappears (and the current row highlights). 

247 self._refresh_chat_list(chats)

249 async def _select_chat(self, chat_id: str) -> None: 

250 if self._bridge is not None: 

251 self._bridge.stop() 

252 self._bridge = None 

253 self._chat_id = chat_id 

254 self._phase.setText("") 

255 self._composer.set_running(False) 

256 d = await engine.get_chat(chat_id) 

257 if d is None: 

258 return 

259 self._title.setText(d["title"] or "Council session") 

260 self._composer.set_idea(d["idea"] or "") 

261 self._composer.set_documents([doc["filename"] for doc in d["documents"]]) 

262 runs = {r["agent_key"]: r for r in d["runs"]} 

263 active = d.get("active_agents") # None = all members active 

264 for key, panel in self._panels.items(): 

265 panel.set_active(active is None or key in active) 

266 r = runs.get(key) 

267 if r: 

268 panel.hydrate(r["status"], r["output_text"], r["provider"], r["model"], r["error"]) 

269 panel.set_chatted(bool(r.get("chatted"))) 

270 else: 

271 panel.reset() 

272 if d["report_markdown"]: 

273 self._report.set_markdown(d["report_markdown"]) 

274 else: 

275 self._report.reset() 

276 # The Chairman can be convened once deliberation has produced contributions. 

277 self._report.set_convene_enabled(d["status"] in ("deliberated", "completed"))

289 def _start_bridge(self) -> CouncilBridge: 

290 """Subscribe a fresh bridge to the chat's event stream and wire the live-view slots.""" 

291 if self._chat_id is None: 

292 raise RuntimeError("Cannot start a bridge without an active chat.") 

293 if self._bridge is not None: 

294 self._bridge.stop() 

295 bridge = CouncilBridge(self._chat_id) 

296 bridge.agent_started.connect(self._on_started) 

297 bridge.agent_token.connect(self._on_token) 

298 bridge.agent_complete.connect(self._on_complete) 

299 bridge.agent_failed.connect(self._on_failed) 

300 bridge.phase_changed.connect(self._on_phase) 

301 bridge.council_report.connect(self._on_report) 

302 bridge.deliberation_complete.connect(self._on_deliberation_complete) 

303 bridge.error.connect(self._on_error) 

304 bridge.finished.connect(lambda: _spawn(self._on_finished())) 

305 bridge.start() 

306 self._bridge = bridge 

307 return bridge

339 def _on_started(self, key: str, provider: str, model: str) -> None: 

340 self._running_agents.add(key)

343 dlg = self._live_dialog_for(key) 

344 if dlg is not None:

347 def _on_token(self, key: str, delta: str) -> None: 

348 if key in self._panels: 

349 self._panels[key].append_token(delta) 

350 dlg = self._live_dialog_for(key) 

351 if dlg is not None:

354 def _on_complete(self, key: str, text: str) -> None: 

355 self._running_agents.discard(key) 

356 if key in self._panels: 

357 self._panels[key].set_complete(text) 

358 dlg = self._live_dialog_for(key) 

359 if dlg is not None: 

360 dlg.end_live(text)

362 def _on_failed(self, key: str, error: str) -> None: 

363 self._running_agents.discard(key) 

364 if key in self._panels: 

365 self._panels[key].set_failed(error) 

366 dlg = self._live_dialog_for(key) 

367 if dlg is not None: 

368 dlg.end_live(f"⚠ {error}")

400 async def _open_agent_chat(self, agent_key: str) -> None: 

401 if not self._chat_id: 

402 return 

403 chat_id = self._chat_id 

404 member = next((m for m in engine.registry().members if m.key == agent_key), None) 

405 # Fetch the conversation on the live loop, then render synchronously — qasync does not 

406 # pump asyncio inside a modal exec()'s nested loop, so we must not load there. 

407 data = await engine.agent_conversation(self._chat_id, agent_key) 

408 if data is None: 

409 data = {"agent_name": member.name if member else agent_key, "turns": []} 

410 dialog = AgentChatDialog(self, self._chat_id, agent_key, data) 

411 # Modal to the window but NON-blocking (show, not exec) so streamed replies keep running. 

412 dialog.setWindowModality(Qt.WindowModality.ApplicationModal)

414 def _on_closed(_r) -> None: 

415 if self._chat_dialog is dialog: 

416 self._chat_dialog = None 

417 self._chat_dialog_key = None 

418 _spawn(self._select_chat(chat_id)) 

419 

420 dialog.finished.connect(_on_closed) 

421 self._chat_dialog = dialog # keep a reference so it isn't garbage-collected 

422 self._chat_dialog_key = agent_key 

423 dialog.show() 

424 # If this agent is mid-deliberation, seed + stream its in-progress response live. 

425 if agent_key in self._running_agents: 

426 panel = self._panels.get(agent_key) 

427 dialog.begin_live(panel.current_text() if panel else "")
