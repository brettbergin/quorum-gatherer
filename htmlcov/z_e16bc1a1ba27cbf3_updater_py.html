1"""Self-update via tufup: heartbeat check -> background download -> apply + restart. 

2 

3Only active in a frozen (installed) build. TUF metadata is served from GitHub Pages and 

4the (large) target archives from GitHub Releases. The trusted public root metadata ships 

5inside the app bundle; if it is absent (repo not yet initialized), updating is a no-op. 

6""" 

7 

8from __future__ import annotations 

9 

10import os 

11import platform 

12import shutil 

13import sys 

14from pathlib import Path 

15 

16BASE_APP_NAME = "quorum-gatherer" 

17GH = "brettbergin/quorum-gatherer" 

18 

19 

20def _os_tag() -> str: 

21 return {"Darwin": "macos", "Windows": "windows", "Linux": "linux"}.get( 

22 platform.system(), "other" 

23 ) 

24 

25 

26# Per-platform TUF streams: tufup names target archives by app+version only, so each OS 

27# needs its own app name + metadata path to avoid collisions on the shared targets release. 

28APP_NAME = f"{BASE_APP_NAME}-{_os_tag()}" 

29 

30METADATA_BASE_URL = os.environ.get( 

31 "QUORUM_UPDATE_METADATA_URL", 

32 f"https://brettbergin.github.io/{BASE_APP_NAME}/metadata/{_os_tag()}/", 

33) 

34TARGET_BASE_URL = os.environ.get( 

35 "QUORUM_UPDATE_TARGET_URL", 

36 f"https://github.com/{GH}/releases/download/updates/", 

37) 

38 

39 

40def is_frozen() -> bool: 

41 return bool(getattr(sys, "frozen", False)) 

42 

43 

44def _bundled_root() -> Path | None: 

45 """The public root.json shipped in the bundle (data file at tuf/root.json).""" 

46 base = Path(getattr(sys, "_MEIPASS", Path(__file__).resolve().parents[1])) 

47 candidate = base / "tuf" / "root.json" 

48 return candidate if candidate.exists() else None 

49 

50 

51def _install_dir() -> Path: 

52 # Directory that holds the installed app (parent of the executable / .app bundle). 

53 exe = Path(sys.executable).resolve() 

54 if sys.platform == "darwin" and ".app/Contents/MacOS" in str(exe): 

55 # .../Foo.app/Contents/MacOS/exe -> install dir is the folder containing Foo.app 

56 return exe.parents[3] 

57 return exe.parent 

58 

59 

60def _make_client(current_version: str): 

61 from tufup.client import Client 

62 

63 from quorum_desktop.paths import data_dir 

64 

65 base = data_dir() / "tuf" 

66 metadata_dir = base / "metadata" 

67 target_dir = base / "targets" 

68 metadata_dir.mkdir(parents=True, exist_ok=True) 

69 target_dir.mkdir(parents=True, exist_ok=True) 

70 

71 # Seed the trusted root on first run from the bundled public root metadata. 

72 root = metadata_dir / "root.json" 

73 if not root.exists(): 

74 bundled = _bundled_root() 

75 if bundled is None: 

76 return None # repo not initialized yet -> updates disabled 

77 shutil.copy(bundled, root) 

78 

79 return Client( 

80 app_name=APP_NAME, 

81 app_install_dir=str(_install_dir()), 

82 current_version=current_version, 

83 metadata_dir=str(metadata_dir), 

84 metadata_base_url=METADATA_BASE_URL, 

85 target_dir=str(target_dir), 

86 target_base_url=TARGET_BASE_URL, 

87 refresh_required=False, 

88 ) 

89 

90 

91def check_and_apply(current_version: str, *, apply: bool = True) -> bool: 

92 """Return True if an update was found (and applied/restarted when apply=True). 

93 

94 A no-op outside a frozen build or before the TUF repo exists. 

95 """ 

96 if not is_frozen(): 

97 return False 

98 client = _make_client(current_version) 

99 if client is None: 

100 return False 

101 new_update = client.check_for_updates() 

102 if not new_update: 

103 return False 

104 if apply: 

105 # Downloads, verifies, applies, and relaunches the app. 

106 client.download_and_apply_update(skip_confirmation=True) 

107 return True

110def updates_supported() -> bool: 

111 """True when this build can self-update (frozen + TUF root present).""" 

112 return is_frozen() and _make_client("0.0.0") is not None 

113 

114 

115def available_update(current_version: str) -> str | None: 

116 """Return the available newer version string, or None. Blocking (run off the UI thread).""" 

117 if not is_frozen(): 

118 return None 

119 client = _make_client(current_version) 

120 if client is None: 

121 return None 

122 new_update = client.check_for_updates() 

123 if not new_update: 

124 return None 

125 version = getattr(new_update, "version", None) 

126 return str(version) if version else str(new_update) 

127 

128 

129def apply_update(current_version: str) -> bool: 

130 """Download + apply the latest update and relaunch. Blocking (run off the UI thread).""" 

131 return check_and_apply(current_version, apply=True)

134def schedule_heartbeat(parent, current_version: str, interval_hours: float = 6.0) -> None: 

135 """Check shortly after launch and then on a cadence (frozen builds only).""" 

136 if not is_frozen(): 

137 return 

138 import contextlib 

139 

140 from PySide6.QtCore import QTimer 

141 

142 def tick() -> None: 

143 with contextlib.suppress(Exception): # never let an update check crash the app 

144 check_and_apply(current_version) 

145 

146 QTimer.singleShot(5_000, tick) 

147 timer = QTimer(parent) 

148 timer.timeout.connect(tick) 

149 timer.start(int(interval_hours * 3600 * 1000)) 

150 parent._update_timer = timer # keep a reference
